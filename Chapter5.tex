\paragraph{So sánh và đánh giá:}

\begin{itemize}

\item \textbf{Forward Chaining:}
\begin{itemize}
    \item Thời gian chạy rất thấp (0.089s), bộ nhớ sử dụng nhỏ (3.05MB), và chỉ mở rộng 1 node.
    \item Tuy nhiên, không tìm được nghiệm.
    \item Điều này cho thấy phương pháp suy luận tiến chỉ thực hiện propagation ràng buộc nhưng không có cơ chế tìm kiếm, do đó thất bại khi bài toán yêu cầu khám phá không gian trạng thái sâu hơn.
\end{itemize}

\item \textbf{A* với Heuristic 1:}
\begin{itemize}
    \item Bị timeout sau 300s, mở rộng 182,217 nodes và sử dụng bộ nhớ lớn (1905.46MB).
    \item Hiệu năng kém cho thấy heuristic 1 không đủ mạnh để định hướng tìm kiếm, dẫn đến việc mở rộng nhiều trạng thái không cần thiết.
\end{itemize}

\item \textbf{A* với Heuristic 2:}
\begin{itemize}
    \item Cũng bị timeout, nhưng số node mở rộng giảm đáng kể (80,879 nodes) và bộ nhớ thấp hơn (325.93MB) so với Heuristic 1.
    \item Điều này chứng tỏ heuristic 2 có khả năng định hướng tốt hơn, giúp giảm không gian tìm kiếm.
    \item Tuy nhiên, vẫn chưa đủ hiệu quả để giải bài toán trong giới hạn thời gian.
\end{itemize}

\item \textbf{Brute Force:}
\begin{itemize}
    \item Mở rộng số lượng node cực lớn (2,428,877 nodes) và bị timeout.
    \item Bộ nhớ sử dụng thấp (0.15MB) do không lưu nhiều trạng thái.
    \item Đây là phương pháp kém hiệu quả nhất về mặt thời gian do không có bất kỳ cơ chế cắt tỉa nào.
\end{itemize}

\item \textbf{Backtracking:}
\begin{itemize}
    \item Tương tự brute force nhưng có cải thiện nhẹ nhờ cắt tỉa một phần.
    \item Tuy nhiên vẫn mở rộng hơn 2 triệu nodes và vượt quá thời gian giới hạn (314.86s).
    \item Bộ nhớ rất cao (10589.39MB), cho thấy chi phí lưu trạng thái và recursion stack lớn.
\end{itemize}

\end{itemize}

\textbf{So sánh tổng thể:}

\begin{itemize}
    \item Về \textbf{khả năng giải bài}: không có phương pháp nào tìm được nghiệm trong thời gian giới hạn.
    
    \item Về \textbf{hiệu quả heuristic}:
    \begin{itemize}
        \item Heuristic 2 tốt hơn Heuristic 1 (ít node hơn, ít bộ nhớ hơn).
        \item Tuy nhiên cả hai vẫn chưa đủ mạnh để xử lý testcase khó này.
    \end{itemize}
    
    \item Về \textbf{chi phí tìm kiếm}:
    \begin{itemize}
        \item Brute Force và Backtracking có chi phí rất lớn (hàng triệu node).
        \item A* giúp giảm đáng kể số node nhưng vẫn chưa đủ.
    \end{itemize}
    
    \item Về \textbf{suy luận ràng buộc}:
    \begin{itemize}
        \item Forward Chaining rất nhanh nhưng không đủ để giải bài toán khi thiếu bước tìm kiếm.
    \end{itemize}

\end{itemize}

\end{itemize}

\begin{itemize}
    \item Là phương pháp nhanh nhất (0.0045s), với 132 nodes.

\end{itemize}

\end{itemize}

    \item Về \textbf{hiệu quả heuristic}:
    \begin{itemize}
        \item Heuristic 2 vượt trội so với Heuristic 1.
        \item Heuristic 1 không phù hợp trong trường hợp thiếu ràng buộc.
    \end{itemize}
    
    \item Về \textbf{chi phí tìm kiếm}:
    \begin{itemize}
        \item Brute Force và Backtracking có chi phí rất thấp do dừng sớm.

    \item Về \textbf{ảnh hưởng của ràng buộc}:
    \begin{itemize}
        \item Việc không có constraint khiến bài toán trở nên “dễ” theo nghĩa tìm nghiệm đầu tiên.
        \item Tuy nhiên, heuristic trở nên kém hiệu quả do thiếu thông tin định hướng.
    \end{itemize}

\end{itemize}
